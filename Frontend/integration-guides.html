WordPress Integration Guide

Follow this step-by-step guide to integrate Transalo into your WordPress site

1

Install the Plugin

Download and install the Transalo WordPress plugin from the WordPress admin panel.

2

Configure API Key

Add your Transalo API key in the plugin settings under Settings > Transalo.

3

Select Languages

Choose which languages you want to support and configure language switcher.

4

Test Integration

Test the integration by switching languages and verifying translations.

WordPress Configuration

// Add to functions.php
add_action('wp_enqueue_scripts', 'transalo_enqueue_scripts');

function transalo_enqueue_scripts() {
    wp_enqueue_script(
        'transalo-script',
        plugin_dir_url(__FILE__) . 'js/transalo.js',
        array('jquery'),
        '1.0.0',
        true
    );
    
    wp_localize_script('transalo-script', 'transalo_ajax', array(
        'ajax_url' => admin_url('admin-ajax.php'),
        'nonce' => wp_create_nonce('transalo_nonce')
    ));
}

// Add language switcher
function transalo_language_switcher() {
    $languages = get_option('transalo_languages', array());
    $current_lang = get_option('transalo_current_language', 'en');
    
    echo '
';
    foreach ($languages as $lang) {
        $active = ($lang === $current_lang) ? 'active' : '';
        echo '' . strtoupper($lang) . '';
    }
    echo '
';
}
